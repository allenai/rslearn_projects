- basecfg.yaml + (basecfg_swinb.yaml | basecfg_swinb_ts.yaml | basecfg_swinb_mm.yaml) (random)

- basecfg.yaml + (basecfg_swinb.yaml | basecfg_swinb_ts.yaml) + satlaspretrain_swinb.yaml (SatlasPretrain)
- basecfg.yaml + (basecfg_helios.yaml | basecfg_helios_ts.yaml | basecfg_helios_mm.yaml) + helios_freeze_then_lowlr.yaml (Helios)
